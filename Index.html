    <script>

        // DOM Elements

        const monthlyInput = document.getElementById('monthlyInput');

        const monthlyRange = document.getElementById('monthlyRange');

        const stepUpInput = document.getElementById('stepUpInput');

        const stepUpRange = document.getElementById('stepUpRange');

        const rateInput = document.getElementById('rateInput');

        const rateRange = document.getElementById('rateRange');

        const yearsInput = document.getElementById('yearsInput');

        const yearsRange = document.getElementById('yearsRange');


        const maturityValueDisp = document.getElementById('maturityValue');

        const investedLabel = document.getElementById('investedLabel');

        const returnsLabel = document.getElementById('returnsLabel');

        const investedBar = document.getElementById('investedBar');

        const returnsBar = document.getElementById('returnsBar');

        const finalMonthlyDisp = document.getElementById('finalMonthlyDisp');

        const totalInvestedMobile = document.getElementById('totalInvestedMobile');

        const wealthGainedMobile = document.getElementById('wealthGainedMobile');

        const projectionTable = document.getElementById('projectionTable');

        const tableSection = document.getElementById('tableSection');


        // Initial Calculation

        updateCalculations();


        // Event Listeners for Syncing Range & Inputs

        function setupSync(input, range) {

            input.addEventListener('input', () => {

                range.value = input.value;

                updateCalculations();

            });

            range.addEventListener('input', () => {

                input.value = range.value;

                updateCalculations();

            });

        }


        setupSync(monthlyInput, monthlyRange);

        setupSync(stepUpInput, stepUpRange);

        setupSync(rateInput, rateRange);

        setupSync(yearsInput, yearsRange);


        function updateCalculations() {

            const P = parseFloat(monthlyInput.value) || 0;

            const stepUpPercent = parseFloat(stepUpInput.value) || 0;

            const annualRate = parseFloat(rateInput.value) / 100;

            const years = parseFloat(yearsInput.value) || 0;

            const monthlyRate = annualRate / 12;


            let totalValue = 0;

            let totalInvested = 0;

            let currentMonthly = P;

            

            const yearlyData = [];


            for (let y = 1; y <= years; y++) {

                // Monthly calculation for the year

                for (let m = 1; m <= 12; m++) {

                    totalInvested += currentMonthly;

                    // Compound existing value + add new investment with immediate month's interest

                    totalValue = (totalValue + currentMonthly) * (1 + monthlyRate);

                }

                

                // Store year end snapshot

                yearlyData.push({

                    year: y,

                    monthly: currentMonthly,

                    invested: totalInvested,

                    total: totalValue

                });


                // Apply Step-up for next year

                currentMonthly = currentMonthly * (1 + (stepUpPercent / 100));

            }


            // Results from final month

            const finalMaturity = totalValue;

            const finalInvested = totalInvested;

            const finalReturns = finalMaturity - finalInvested;

            const lastYearMonthly = yearlyData.length > 0 ? yearlyData[yearlyData.length - 1].monthly : P;


            // Update UI Labels (Removed currency symbols, used plain format)

            const formatter = new Intl.NumberFormat('en-US', {

                maximumFractionDigits: 0

            });


            maturityValueDisp.textContent = formatter.format(finalMaturity);

            investedLabel.textContent = `Invested: ${formatter.format(finalInvested)}`;

            returnsLabel.textContent = `Returns: ${formatter.format(finalReturns)}`;

            finalMonthlyDisp.textContent = formatter.format(lastYearMonthly);

            totalInvestedMobile.textContent = formatter.format(finalInvested);

            wealthGainedMobile.textContent = formatter.format(finalReturns);


            // Update Visualization Bars

            const investedPercent = (finalInvested / finalMaturity) * 100 || 0;

            const returnsPercent = (finalReturns / finalMaturity) * 100 || 0;

            

            investedBar.style.width = `${investedPercent}%`;

            returnsBar.style.width = `${returnsPercent}%`;


            renderTable(yearlyData);

        }


        function renderTable(data) {

            projectionTable.innerHTML = '';

            data.forEach(item => {

                const row = document.createElement('tr');

                row.className = 'hover:bg-slate-50 transition-colors';

                row.innerHTML = `

                    <td class="px-4 py-3 border-b border-slate-100 font-medium text-slate-700">Year ${item.year}</td>

                    <td class="px-4 py-3 border-b border-slate-100 text-slate-500">${formatCompact(item.monthly)}</td>

                    <td class="px-4 py-3 border-b border-slate-100 text-slate-600">${formatCompact(item.invested)}</td>

                    <td class="px-4 py-3 border-b border-slate-100 font-bold text-slate-800">${formatCompact(item.total)}</td>

                `;

                projectionTable.appendChild(row);

            });

        }


        function formatCompact(val) {

            return new Intl.NumberFormat('en-US', {

                notation: 'compact',

                maximumFractionDigits: 1

            }).format(val);

        }


        function toggleTable() {

            if (tableSection.classList.contains('hidden')) {

                tableSection.classList.remove('hidden');

            } else {

                tableSection.classList.add('hidden');

            }

        }
